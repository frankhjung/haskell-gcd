   15 module Gcd (euclid1, euclid2) where
   16 
   17 -- | Greatest Common Denominator (for numbers greater than 0)
   18 
   19 -- | Method 1 - using only subtraction
   20 euclid1 :: Word -> Word -> Word
   21 euclid1 u v | u <= 0    = 0
   22             | v <= 0    = 0
   23             | u == v    = u
   24             | u < v     = euclid1 u (v - u)
   25             | otherwise = euclid1 (u - v)  v
   26 
   27 -- | Method 2 - using modulus
   28 euclid2 :: Word -> Word -> Word
   29 euclid2 u v = if remainder == 0
   30                 then v
   31                 else euclid2 v remainder
   32               where remainder = u `mod` v
